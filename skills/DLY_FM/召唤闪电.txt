var 魔力保护 = 200;
if(自己.魔 < 魔力保护) {
  return false;
}

// 找一个范围内最远的打
var 攻击目标 = 0;
if(获得表数量(怪) > 0) {
  排序(怪,"距离")
  for(i=1;获得表数量(怪);1) {
    if(怪[i].距离 < 技能距离) {
      攻击目标 = 怪[i].id;
    } else {
      break;
    }
  }
}

// 没目标
if(攻击目标 == 0) {
  return false;
}

var bsuccess = 使用技能(技能id,技能等级,攻击目标);

// 没成功，选自己放
if(!bsuccess) {
  攻击目标 = 自己.id
  bsuccess = 使用技能(技能id,技能等级,攻击目标);
}

return bsuccess,false;
