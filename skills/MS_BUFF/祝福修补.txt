
var 失血保护 = 600;
var 照顾队友 = true;

// 坐下时不使用技能
if(自己.状态 != 5) {
  return false;
}

var 技能目标 = {};

if(照顾队友) {
  for(i=1;获得表数量(队友);1) {
    if(队友[i].距离 < 技能距离) {
      添加进表(技能目标,队友[i]);
    }
  }
}

添加进表(技能目标,自己);

var 失血量 = 0;
var 最大失血量 = 0;
var 需要加血的目标 = 0;

for(i=1;获得表数量(技能目标);1) {
  失血量 = 技能目标[i].最大血 - 技能目标[i].血;
  if(失血量 > 失血保护 && 失血量 > 最大失血量) {
    需要加血的目标 = 技能目标[i].id;
    最大失血量 = 失血量;
  }
}

if(需要加血的目标 > 0) {
  return 使用技能(技能id,技能等级,需要加血的目标);
}

return false;
