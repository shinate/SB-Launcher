//根据队伍成员的变化进行群体buff判断

// 坐下时不使用技能
if(自己.状态 != 5) {
  return false;
}

var 持续时间 = 10800000;
var 队伍锁定名 = 技能名;

排序(队友,"名字");

for(i=1;获得表数量(队友);1) {
  if(队友[i].距离 < 技能距离) {
    队伍锁定名 += "-" + 队友[i].名字;
  }
}

// 如果队伍成员发生变化，解锁自身
if(0 == 锁定技能(自己.id,队伍锁定名,持续时间)) {
  锁定技能(自己.id,技能名,0);
}

if(0 < 锁定技能(自己.id,技能名,持续时间)) {
  return false;
}

return 使用技能(技能id,技能等级,自己.id);
